> **주의 2. | 와 || **
>
> ```c++
> // 합쳐지는 파이어볼의 방향이 모두 홀수이거나 모두 짝수이면, 방향은 0, 2, 4, 6이 되고, 그렇지 않으면 1, 3, 5, 7이 된다.
> 
> bool Sdcheck=false;
> Sdcheck = Sdcheck || ( 이전의값의 나머지 != 이전값의 나머지 );
> 
> //또는
> int  Sdcheck = 0;
> Sdcheck |= (  이전의값의 나머지 != 이전값의 나머지 ) 
> 
> // 오류
> int Sdcheck;
> Sdcheck = Sdcheck | ( 이전의값의 나머지 != 이전값의 나머지 );
> 
> ```
